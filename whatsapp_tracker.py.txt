# Endpoint to handle incoming requests from the shared link
@app.route('/track', methods=['GET'])
def track_location():
    # Extract query parameters
    phone_number = request.args.get('phone')
    ip_address = request.args.get('ip')

    if not phone_number or not ip_address:
        return jsonify({'error': 'Missing phone number or IP address'}), 400

    # Get geolocation data
    geolocation_data = get_geolocation(ip_address)
    if geolocation_data:
        return jsonify({
            'phone_number': phone_number,
            'geolocation': geolocation_data
        })
    else:
        return jsonify({'error': 'Unable to retrieve geolocation data'}), 500
